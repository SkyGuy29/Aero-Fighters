362void Player::update(const sf::Vector2f winSize, std::vector<Object*>* objects, bool time)
363{
364 setSize(20, 32);
365 //If the player is currently dead, don't do anything
366 if (timerDeath > 0)
367 {
368 setSize(0, 0);
369 timerDeath--;
370 return;
371 }
372 //If gone do nothing
373 if (health == 0)
374 return;
375 //Am I being shot?
376 for (unsigned int i = 0; i < objects->size(); i++)
377 {
378 if (((objects->at(i)->getType() == ENEMY_PROJECTILE)
379 || (objects->at(i)->getType() == AIR))
380 && this->intersect(objects->at(i))
381 && !invincibility)
382 {
383 //If the player is hit while not invincible
384 health--;
385 objects->push_back(new Explosion(pos, 0));
386 objects->push_back(new Collectable(1, pos, backgroundSpeed));
387 specialCharge = 2;
388 if (rand() % 5 == 0)
389 objects->push_back(new Collectable(2, pos, backgroundSpeed));
390 if (!health)
391 {
392 if(rand() % 25 == 0)
393 objects->push_back(new Collectable(3, pos,
394 backgroundSpeed));
395 }
396 pos = sf::Vector2f(winSize.x * 0.5f, winSize.y * 0.75f);
397 invincibility = 61;
398 timerDeath = 61;
399 setSize(0, 0);
400 }
401 else if (objects->at(i)->getType() == COLLECTABLE
402 && this->intersect(objects->at(i)))
403 {
404 switch (objects->at(i)->getId())
405 {
406 case 0: //Interact with money
407 //Increase score
408 break;
409 case 1: //Interact with P
410 if (powerLevel < 3)
411 powerLevel++;
412 else
413 {
414 //score increase by 2000
415 }
416 break;
417 case 2: //Interact with B
418 if (specialCharge < 5)
419 specialCharge++;
420 break;
421 case 3: //Interact with F
422 powerLevel = 3;
423 break;
424 }
425 }
426 }
427
428 if (cooldown != 0)
429 cooldown--;
430
431 if (cooldownSecondary != 0)
432 cooldownSecondary--;
433
434 if (invincibility != 0)
435 invincibility--;
436
437 if (timer > 0)
438 {
439 timer--;
440 if (timer <= 0)
441 movingProjectile = nullptr;
442 }
443
444 if (cooldownTime != 0)
445 cooldownTime--;
446
447 Player::move(winSize);
448
449 if (movingProjectile != nullptr)
450 movingProjectile->setPos(pos.x, pos.y - winSize.y / 2);
451}
