69void Object::nextFrame(const int frameRate = 15)
70{
71 // Increases the image rectangle by its height and loops back when
72 //it reaches the end
73 currentFrame++;
74 if (currentFrame >= frameCount * frameRate)
75 {
76 currentFrame -= frameCount * frameRate;
77 anDone = true;
78 }
79
80 // the dividing to an int is needed for the updates per frame delay.
81 sprite.setTextureRect(sf::IntRect(
82 texOffset.x + (currentFrame / frameRate) * texSize.x
83 * !verticalAnimation,
84 texOffset.y + (currentFrame / frameRate) * texSize.y
85 * verticalAnimation,
86 texSize.x, texSize.y));
87}
